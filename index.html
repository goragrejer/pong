// --- DOM References --- const color1 = document.getElementById('color1'); const color2 = document.getElementById('color2'); const gradientBox = document.getElementById('gradient-box'); const cssOutput = document.getElementById('css-output'); // --- Core Logic --- /** * Generates the CSS gradient string and updates the UI. */ function generateGradient() { // 1. Get the current colors const c1 = color1.value; const c2 = color2.value; // 2. Create the linear-gradient CSS value // We use a 45-degree angle for that dynamic, diagonal look const gradientValue = `linear-gradient(45deg, ${c1}, ${c2})`; // 3. Update the Live Preview Box gradientBox.style.background = gradientValue; // 4. Update the CSS Output Text Area // Provide the full CSS block, ready to copy-paste into style.css cssOutput.value = `/* Copy and Paste this block into your style.css */ .my-custom-card { background: ${gradientValue}; /* Don't forget to define a solid border! */ border: 1px solid ${c1}; }`; } // --- Event Listeners --- // Update the gradient whenever a color picker value changes color1.addEventListener('input', generateGradient); color2.addEventListener('input', generateGradient); // Initial call to set the default gradient when the page loads generateGradient();
